create table request(
    request_id int auto_increment not null comment '主キー', 
    user_id int(11) not null,
    experiment_num varchar(50) not null default '' comment '実験番号',
    titlename text not null,
    dnaflag boolean not null,
    dna_approval_num varchar(50),
    humanflag boolean not null,
    human_approval_num varchar(50),
    drugflag boolean not null,
    drug_name varchar(255),
    resercher_name varchar(50),
    licence_num varchar(50),
    purpose text not null,
    outline text not null,
    start_time datetime not null,
    finish_time datetime not null,
    reason varchar(50) not null,
    other_reason varchar(50),
    category varchar(50) not null,
    format1 varchar(50),
    content_num1 varchar(50),
    format2 varchar(50),
    content_num2 varchar(50),
    format3 varchar(50),
    content_num3 varchar(50),
    format_drug_name varchar(50),
    method_name varchar(50), 
    format4 varchar(50),
    content_num4 varchar(50),
    format5 varchar(50),
    content_num5 varchar(50),
    format6 varchar(50),
    content_num6 varchar(50),
    treatment1 varchar(50),
    treatment_drug_name1 varchar(255),
    treatment_method_name1 varchar(50),
    treatment2 varchar(50),
    treatment_drug_name2 varchar(255),
    treatment_method_name2 varchar(50),
    treatment3 varchar(50),
    gas_name varchar(50),
    treatment4 varchar(50),
    treatment5 varchar(50),
    treatment6 varchar(50), 
    other_treatment varchar(50),
    remarks text,
    zip_file varchar(50),
    allow boolean not null default '0' comment '承認済みか否か',
    request_date varchar(50) not null comment '申請日',
    index(request_id),
    primary key (request_id),
    foreign key (user_id)
    references Users(user_id)
) engine=innodb default charset=utf8mb4;

create table Place(
    place_id int(11) auto_increment not null comment '実験場所の主キー',
    request_id int not null,
    animal varchar(20) not null,
    strain_name varchar(50) not null,
    male varchar(10) not null,
    female varchar(10) not null,
    age varchar(50) not null,
    buy varchar(50),
    out_distribution varchar(50),
    in_distribution varchar(50),
    breeding varchar(50),
    total varchar(50),
    quality varchar(50) not null,
    place varchar(50) ,
    other_place varchar(50),
    other_place_text varchar(100),
    index(place_id),
    primary key (place_id),
    foreign key (request_id)
    references request(request_id)
) engine=innodb default charset=utf8mb4;

create table Users(
    user_id int(11) auto_increment not null comment 'ユーザー作成時に自動生成される番号　主キー',  
    account_id varchar(40) not null default '' comment 'ログイン時に入力するユーザー名',
    user_password varchar(255) not null default '' comment 'パスワード　sha256でhash化',
    user_name varchar(40) not null comment 'ユーザーのフルネーム', 
    pronunciation varchar(50) not null comment 'ユーザーのフリガナ',
    mailAdress varchar(255) not null, 
    belong varchar(100) not null default '',
    admin_user boolean not null default '0' comment '管理者であるか否か',
    chief_user boolean not null default '0' comment '所長であるか否か', 
    index(user_id),
    primary key (user_id),
    unique key (account_id)
) engine=innodb default charset=utf8mb4;

create table session(
    session_id int(11) auto_increment not null comment '主キー',
    user_id int(11) not null comment 'クッキー作成時のユーザー番号　Usersのuser_idと同一',
    session_key varchar(255) not null default '' comment 'クッキーを使ってブラウザに保存するsessionキー',
    index(session_id),
    primary key (session_id),
    unique key (user_id),
    foreign key (user_id)
    references Users(user_id)
) engine=innodb default charset=utf8mb4;
